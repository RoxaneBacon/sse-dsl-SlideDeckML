Code Blocks with Syntax Highlighting

JavaScript Example

function fibonacci(n) {
    if (n <= 1) return n;
    return fibonacci(n - 1) + fibonacci(n - 2);
}

const result = fibonacci(10);
console.log(`Fibonacci(10) = ${result}`);

Python with Line Numbers

def greet(name):
    message = f"Hello, {name}!"
    print(message)
    print("Welcome to SlideDeckML!")
    return message

greet("World")

TypeScript with Custom Line Start

interface Presentation {
    title: string;
    author: string;
    slides: Slide[];
}

class SlideDeck implements Presentation {
    constructor(
        public title: string,
        public author: string,
        public slides: Slide[]
    ) {}
}
